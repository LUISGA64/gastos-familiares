# Comandos Útiles para Railway Deploy

# 1. INICIALIZAR GIT (si no lo has hecho)
git init
git add .
git commit -m "Preparado para deploy en Railway"

# 2. CONECTAR CON GITHUB
git remote add origin https://github.com/TU_USUARIO/gastos-familiares.git
git branch -M main
git push -u origin main

# 3. ACTUALIZAR CÓDIGO (después de hacer cambios)
git add .
git commit -m "Descripción de tus cambios"
git push

# 4. GENERAR SECRET_KEY SEGURO (Python)
# Ejecuta esto en Python para generar una nueva SECRET_KEY:
# python -c "from django.core.management.utils import get_random_secret_key; print(get_random_secret_key())"

# 5. VARIABLES DE ENTORNO EN RAILWAY
# Copia estas variables en Railway > Variables:
#
# SECRET_KEY=genera-una-nueva-con-el-comando-de-arriba
# DEBUG=False
# ALLOWED_HOSTS=*.railway.app
# AI_PROVIDER=groq
# GROQ_API_KEY=tu-groq-api-key

# 6. COMANDOS EN RAILWAY SHELL
# Desde Railway > Settings > Deploy Logs > ... > Create Shell:
#
# Crear superusuario:
# python manage.py createsuperuser
#
# Ejecutar migraciones manualmente:
# python manage.py migrate
#
# Recolectar archivos estáticos:
# python manage.py collectstatic --noinput
#
# Abrir shell de Django:
# python manage.py shell

# 7. VER LOGS EN RAILWAY
# Railway > Deployments > Click en el deploy activo
# Los logs se actualizan en tiempo real

# 8. ROLLBACK A VERSIÓN ANTERIOR
# Railway > Deployments > Click en un deploy antiguo > Redeploy

# 9. COMANDOS GIT ÚTILES
# Ver estado:
git status

# Ver commits:
git log --oneline

# Deshacer último commit (mantiene cambios):
git reset --soft HEAD~1

# Deshacer cambios en archivo:
git checkout -- archivo.py

# 10. RAILWAY CLI (Opcional)
# Instalar:
# npm install -g @railway/cli
#
# Login:
# railway login
#
# Link proyecto:
# railway link
#
# Ver logs:
# railway logs
#
# Ejecutar comando:
# railway run python manage.py createsuperuser
